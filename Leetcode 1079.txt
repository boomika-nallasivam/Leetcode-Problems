class Solution {
    Set<String> s = new HashSet<>();
    public int numTilePossibilities(String tiles) {
        boolean[] visit = new boolean[tiles.length()];
        backtrack(tiles, "", visit);
        return s.size();
    }
    private void backtrack(String tiles, String word, boolean[] visit) {
        for (int i = 0; i < tiles.length(); i++) {
            if (visit[i]) continue;
            String charSequence = word + tiles.charAt(i);
            visit[i] = true;
            s.add(charSequence);
            backtrack(tiles, charSequence, visit);
            visit[i] = false;
        }
    }
}
